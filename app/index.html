set up gulp CLI and dev gulp 4.0

# uninstall previous gulp installation and related packages (if applicable)
# to avoid sudo spam, prefix npm with sudo if you're not executing as root
$ npm rm gulp -g
$ npm rm gulp-cli -g
$ cd your-project-dir/
$ npm rm gulp --save-dev
$ npm rm gulp --save
$ npm rm gulp --save-optional
$ npm cache clean

# install the latest gulp cli tools (globally)
$ npm install gulpjs/gulp-cli -g

# install gulp 4 into your project from 4.0 gitHub branch as dev dependency
$ npm install gulpjs/gulp#4.0 --save-dev
